#include <stdio.h>

int add(int a,int b);
int sub(int a,int b);
int mul(int a,int b);

int main()
{
	printf("李四: %d\r\n",add(7+8));
	
	printf("%d\r\n",sub(6,9));
	printf("%d\r\n",mul(4,5));
	return 0;
}

//最终讨论得到了改函数
int add(int a,int b)
{
	return a+b;
}

//张三
int sub(int a,int b)
{
	return a-b;
}

//张三乘法
int mul(int a,int b)
{
	return a*b;
}
